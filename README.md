## Install

```bash
npm install --save aix-intranet-components
```

## Usage

```tsx
import * as React from 'react'

import MyComponent from 'aix-intranet-components'

class Example extends React.Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
